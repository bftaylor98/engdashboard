This document outlines a solution for connecting to a ProShop instance's GraphQL API to read data for Work Orders and Non-Conformance Reports. It includes Python scripts designed to be user-friendly for Windows environments.

Project Overview
The primary goal is to establish a read-only connection to a cloud-hosted ProShop tenant via its GraphQL API. This connection will be used to retrieve data specifically for Work Orders and Non-Conformance Reports (NCRs). To facilitate this, small, Windows-compatible Python scripts will be provided for the user to run locally.

Authentication
Connection to the ProShop API is managed through a session-based authentication process. A 

POST request to the /api/beginsession endpoint, including the user's credentials, will return a 5-minute access token.  This token must then be included in the headers of subsequent GraphQL API requests for authorization. The scripts will handle this login flow automatically.

Python Scripts for Data Extraction
The following Python scripts have been developed to meet the project's requirements. They are designed to be run on Windows 10/11 with Python 3.x and rely solely on the built-in requests library. As requested, credentials are hard-coded into the scripts for simplicity.

Work Order and Operations Script
This script fetches the first five Work Orders and then retrieves detailed information, including operations, for a specific Work Order (25-0718). It demonstrates how to handle paginated results and access nested data structures within the ProShop GraphQL schema.

import requests
import json

# --- Configuration ---
PROSHOP_TENANT_URL = "https://est.adionsystems.com"
GRAPHQL_ENDPOINT = f"{PROSHOP_TENANT_URL}/api/graphql"
LOGIN_ENDPOINT = f"{PROSHOP_TENANT_URL}/api/beginsession"

# Hard-coded credentials as per user request
USERNAME = "your_username"
PASSWORD = "your_password"

def get_access_token():
    """Authenticates with ProShop and returns a session token."""
    try:
        response = requests.post(LOGIN_ENDPOINT, json={"username": USERNAME, "password": PASSWORD})
        response.raise_for_status()
        return response.json().get("token")
    except requests.exceptions.RequestException as e:
        print(f"Error during authentication: {e}")
        return None

def execute_graphql_query(token, query, variables=None):
    """Executes a GraphQL query against the ProShop API."""
    headers = {"Authorization": f"Bearer {token}"}
    try:
        response = requests.post(GRAPHQL_ENDPOINT, json={"query": query, "variables": variables}, headers=headers)
        response.raise_for_status()
        return response.json()
    except requests.exceptions.RequestException as e:
        print(f"Error executing GraphQL query: {e}")
        return None

def main():
    """Main function to fetch and display Work Order data."""
    token = get_access_token()
    if not token:
        return

    # Query for the first 5 Work Orders
    work_orders_query = """
    query GetWorkOrders($pageSize: Int) {
      workOrders(pageSize: $pageSize) {
        records {
          workOrderNumber
          partNumber
        }
        totalRecords
      }
    }
    """
    print("--- First 5 Work Orders ---")
    wo_data = execute_graphql_query(token, work_orders_query, {"pageSize": 5})
    if wo_data and "data" in wo_data and "workOrders" in wo_data["data"]:
        for wo in wo_data["data"]["workOrders"]["records"]:
            print(f"Work Order: {wo['workOrderNumber']}, Part: {wo['partNumber']}")

    # Query for a specific Work Order with its operations
    specific_wo_query = """
    query GetSpecificWorkOrder($workOrderNumber: String!) {
      workOrders(filters: {workOrderNumber: {NIN: [$workOrderNumber]}}) {
        records {
          workOrderNumber
          partNumber
          ops {
            records {
              opNumber
              opName
              # Other scalar fields can be added here
            }
          }
        }
      }
    }
    """
    print("\n--- Detailed Operations for WO 25-0718 ---")
    specific_wo_data = execute_graphql_query(token, specific_wo_query, {"workOrderNumber": "25-0718"})
    if specific_wo_data and "data" in specific_wo_data and "workOrders" in specific_wo_data["data"]:

            if "ops" in wo and "records" in wo["ops"]:

if __name__ == "__main__":
    main()

Latest Non-Conformance Reports (NCRs) Script
This script demonstrates how to query for the five most recent Non-Conformance Reports. It utilizes GraphQL introspection to dynamically discover the list field name within the nonConformanceReports wrapper, making the script more robust against potential schema variations between tenants.

Python

import requests
import json

# --- Configuration ---
PROSHOP_TENANT_URL = "https://est.adionsystems.com"
GRAPHQL_ENDPOINT = f"{PROSHOP_TENANT_URL}/api/graphql"
LOGIN_ENDPOINT = f"{PROSHOP_TENANT_URL}/api/beginsession"

# Hard-coded credentials
USERNAME = "your_username"
PASSWORD = "your_password"

def get_access_token():
    """Authenticates and returns a session token."""
    try:
        response = requests.post(LOGIN_ENDPOINT, json={"username": USERNAME, "password": PASSWORD})
        response.raise_for_status()
        return response.json().get("token")
    except requests.exceptions.RequestException as e:
        print(f"Error during authentication: {e}")
        return None

def execute_graphql_query(token, query, variables=None):
    """Executes a GraphQL query."""
    headers = {"Authorization": f"Bearer {token}"}
    try:
        response = requests.post(GRAPHQL_ENDPOINT, json={"query": query, "variables": variables}, headers=headers)
        response.raise_for_status()
        return response.json()
    except requests.exceptions.RequestException as e:
        print(f"Error executing GraphQL query: {e}")
        return None

def get_ncr_list_field_name(token):
    """Introspects the schema to find the list field name for NCRs."""
    introspection_query = """
      query IntrospectionQuery {
        __type(name: "NonConformanceReportConnection") {
          fields {
            name
          }
        }
      }
    """
    data = execute_graphql_query(token, introspection_query)
    if data and "data" in data and "__type" in data["data"] and "fields" in data["data"]["__type"]:
        for field in data["data"]["__type"]["fields"]:
            if field["name"].lower() == "list" or field["name"].lower() == "records":
                 return field["name"]
    return "records" # Default fallback

def main():
    """Fetches and displays the latest NCRs."""
    token = get_access_token()
    if not token:
        return

    ncr_list_field = get_ncr_list_field_name(token)

    query = f"""
    query GetLatestNCRs($pageSize: Int) {{
      nonConformanceReports(pageSize: $pageSize) {{
        {ncr_list_field} {{
          ncrNumber
          created
        }}
      }}
    }}
    """
    print("--- Latest 5 Non-Conformance Reports ---")
    data = execute_graphql_query(token, query, {"pageSize": 5})
    if data and "data" in data and "nonConformanceReports" in data["data"]:
        for ncr in data["data"]["nonConformanceReports"][ncr_list_field]:
            print(f"{ncr['ncrNumber']} -> {ncr['created']}")

if __name__ == "__main__":
    main()

## 1. Updated Python Script with Day-of-the-Week Logic
This final version of the script checks the current day. If it's Monday, it uses the "Friday to Monday" time window; otherwise, it uses the existing "yesterday to today" window.

Python

import requests, sys, re, html, json
from datetime import datetime, timezone, timedelta

# ---- CONFIG -----------------------------------------------------
ROOT_URL  = "https://est.adionsystems.com"
USERNAME  = "admin@esttool.com"
PASSWORD  = "[REDACTED - see .env]"
SCOPE     = "nonconformancereports:r workorders:r parts:r"
WEBHOOK_URL = "https://prod-138.westus.logic.azure.com:443/workflows/9a3ae54474864ed3929883d2254861a2/triggers/manual/paths/invoke?api-version=2016-06-01&sp=%2Ftriggers%2Fmanual%2Frun&sv=1.0&sig=Z7iRoHHPuzVUSk5UaHDrmvWj-rQiRK7Dxj-voh8orCU"
# ----------------------------------------------------------------

def begin_session() -> str:
    r = requests.post(f"{ROOT_URL}/api/beginsession", json={"username": USERNAME, "password": PASSWORD, "scope": SCOPE}, headers={"Accept": "application/json"}, timeout=30)
    r.raise_for_status()
    return r.json()["authorizationResult"]["token"]

def gql(query: str, vars_: dict, token: str) -> dict:
    r = requests.post(f"{ROOT_URL}/api/graphql", json={"query": query, "variables": vars_}, headers={"Authorization": f"Bearer {token}", "Accept": "application/json"}, timeout=30)
    body = r.json()
    if "errors" in body:
        raise RuntimeError(body["errors"])
    return body["data"]

def main():
    token = begin_session()

    # MODIFIED: Conditional logic for the time window
    local_tz = timezone(timedelta(hours=-4)) # EDT
    now_local = datetime.now(local_tz)
    
    # weekday() returns 0 for Monday, 1 for Tuesday, etc.
    if now_local.weekday() == 0: # It's Monday
        # Window: Friday at 2:30 PM to Monday at 6:00 AM
        start_time_local = (now_local - timedelta(days=3)).replace(hour=14, minute=30, second=0, microsecond=0)
        end_time_local = now_local.replace(hour=6, minute=0, second=0, microsecond=0)
    else: # It's Tuesday - Friday
        # Window: Yesterday at 4:30 PM to Today at 6:00 AM
        start_time_local = (now_local - timedelta(days=1)).replace(hour=16, minute=30, second=0, microsecond=0)
        end_time_local = now_local.replace(hour=6, minute=0, second=0, microsecond=0)

    # Convert to UTC and format for the API
    start_utc_str = start_time_local.astimezone(timezone.utc).strftime("%Y-%m-%dT%H%M%SZ")
    end_utc_str = end_time_local.astimezone(timezone.utc).strftime("%Y-%m-%dT%H%M%SZ")

    print(f"Fetching NCRs from {start_time_local.strftime('%Y-%m-%d %I:%M %p')} to {end_time_local.strftime('%Y-%m-%d %I:%M %p')}...")

    # The rest of the script remains the same
    data_query = """
    query($startTime: DateTime!, $endTime: DateTime!, $pageSize: Int!){
      nonConformanceReports(
        pageSize: $pageSize,
        filter: { lastModifiedTime: { greaterThanOrEqual: $startTime, lessThanOrEqual: $endTime } }
      ){
        records{ ncrRefNumber, createdTime, lastModifiedByPlainText, notes, workOrder { workOrderNumber, part { partDescription } } }
      }
    }
    """
    variables = { "startTime": start_utc_str, "endTime": end_utc_str, "pageSize": 1000 }
    data = gql(data_query, variables, token)["nonConformanceReports"]["records"]

    if not data:
        print("No recently modified NCRs found."); return

    ncr_payload = []
    for ncr in data:
        time_str = ncr.get("createdTime")
        if not time_str: continue
        utc_time = datetime.strptime(time_str, "%Y-%m-%dT%H%M%SZ").replace(tzinfo=timezone.utc)
        record_created_time_local = utc_time.astimezone(local_tz)
        if start_time_local <= record_created_time_local <= end_time_local:
            notes_text = ncr.get("notes")
            if notes_text:
                clean_text = re.sub(r'<.*?>', '', notes_text); clean_text = html.unescape(clean_text)
                notes_text = clean_text.strip()
            else: notes_text = "N/A"
            work_order = ncr.get("workOrder") or {}; part = work_order.get("part") or {}
            ncr_payload.append({
                "ncrRefNumber": ncr.get("ncrRefNumber", "N/A"), "workOrderNumber": work_order.get("workOrderNumber", "N/A"),
                "partDescription": part.get("partDescription", "No part description."), "createdById": ncr.get("lastModifiedByPlainText", "N/A"),
                "notes": notes_text, "timestamp": record_created_time_local.strftime("%Y-%m-%d %I:%M %p")
            })

    if not ncr_payload: print("No NCRs were both created and modified in the specified time window."); return
    
    final_payload = { "ncr_records": ncr_payload }
    print(f"Sending {len(ncr_payload)} filtered NCR(s) to Power Automate flow...")
    try:
        headers = {'Content-Type': 'application/json'}
        response = requests.post(WEBHOOK_URL, json=final_payload, headers=headers)
        if response.status_code == 202: print("Successfully triggered the Power Automate flow.")
        else: print(f"Failed to trigger flow. Status: {response.status_code}, Response: {response.text}")
    except requests.exceptions.RequestException as e: print(f"Error sending request to webhook: {e}")

if __name__ == "__main__":
    try:
        main()
    except Exception as e:
        print("Error:", e)
        sys.exit(1)

    total_records = count_data['nonConformanceReports']['totalRecords']
    
    # --- Step 2: Fetch ALL records from the API ---
    print(f"Fetching all {total_records} NCRs to filter locally...")
    # This query has no filter and fetches all records by setting pageSize to the total
    data_query = """
    query($pageSize:Int!){
      nonConformanceReports(pageSize: $pageSize){
        records{
          ncrRefNumber, createdTime, lastModifiedByPlainText, notes,
          workOrder { workOrderNumber, part { partDescription } }
        }
      }
    }
    """
    variables = { "pageSize": total_records }
    data = gql(data_query, variables, token)["nonConformanceReports"]["records"]

    if not data:
        print("Could not retrieve any NCRs."); return

    # --- Step 3: Define the time window and filter in Python ---
    local_tz = timezone(timedelta(hours=-4)) # EDT
    now_local = datetime.now(local_tz)
    
    if now_local.weekday() == 0: # Monday
        start_time_local = (now_local - timedelta(days=3)).replace(hour=14, minute=30, second=0, microsecond=0)
        end_time_local = now_local.replace(hour=6, minute=0, second=0, microsecond=0)
    else: # Tuesday - Friday
        start_time_local = (now_local - timedelta(days=1)).replace(hour=16, minute=30, second=0, microsecond=0)
        end_time_local = now_local.replace(hour=6, minute=0, second=0, microsecond=0)

    print(f"Filtering {len(data)} records for window: {start_time_local.strftime('%Y-%m-%d %I:%M %p')} to {end_time_local.strftime('%Y-%m-%d %I:%M %p')}...")

    ncr_payload = []
    for ncr in data:
        created_time_str = ncr.get("createdTime")
        if not created_time_str: continue

        utc_created_time = datetime.strptime(created_time_str, "%Y-%m-%dT%H%M%SZ").replace(tzinfo=timezone.utc)
        local_created_time = utc_created_time.astimezone(local_tz)

        # The precise, local filter based on creation time
        if start_time_local <= local_created_time <= end_time_local:
            notes_text = ncr.get("notes")
            if notes_text:
                clean_text = re.sub(r'<.*?>', '', notes_text); clean_text = html.unescape(clean_text)
                notes_text = clean_text.strip()
            else: notes_text = "N/A"
            work_order = ncr.get("workOrder") or {}; part = work_order.get("part") or {}
            ncr_payload.append({
                "ncrRefNumber": ncr.get("ncrRefNumber", "N/A"),
                "workOrderNumber": work_order.get("workOrderNumber", "N/A"),
                "partDescription": part.get("partDescription", "No part description."),
                "createdById": ncr.get("lastModifiedByPlainText", "N/A"),
                "notes": notes_text,
                "timestamp": local_created_time.strftime("%Y-%m-%d %I:%M %p")
            })

    if not ncr_payload:
        print("No NCRs were created in the specified time window."); return
    
    # --- Step 4: Send the final results ---
    final_payload = { "ncr_records": ncr_payload }
    print(f"Sending {len(ncr_payload)} filtered NCR(s) to Power Automate flow...")
    try:
        headers = {'Content-Type': 'application/json'}
        response = requests.post(WEBHOOK_URL, json=final_payload, headers=headers)
        if response.status_code == 202: print("Successfully triggered the Power Automate flow.")
        else: print(f"Failed to trigger flow. Status: {response.status_code}, Response: {response.text}")
    except requests.exceptions.RequestException as e: print(f"Error sending request to webhook: {e}")

if __name__ == "__main__":
    try:
        main()
    except Exception as e:
        print("Error:", e)
        sys.exit(1)

"""
single_wo_lookup.py
-----------------------------------------------------------
Prints the Work Order number and its Part number
for a single WO you specify (default 25‑0718).
Run:   python single_wo_lookup.py
Needs: pip install requests
-----------------------------------------------------------
"""
import requests, sys

# ---- CONFIG -----------------------------------------------------
ROOT_URL  = "https://est.adionsystems.com"
USERNAME  = "admin@esttool.com"
PASSWORD  = "[REDACTED - see .env]"
SCOPE     = "workorders:r parts:r"

TARGET_WO = "25-0718"   # <— change here if you want another WO
# ----------------------------------------------------------------


def begin_session() -> str:
    r = requests.post(
        f"{ROOT_URL}/api/beginsession",
        json={"username": USERNAME, "password": PASSWORD, "scope": SCOPE},
        headers={"Accept": "application/json"},
        timeout=30,
    )
    r.raise_for_status()
    return r.json()["authorizationResult"]["token"]


def gql(query: str, vars_: dict, token: str) -> dict:
    r = requests.post(
        f"{ROOT_URL}/api/graphql",
        json={"query": query, "variables": vars_},
        headers={"Authorization": f"Bearer {token}",
                 "Accept": "application/json"},
        timeout=30,
    )
    body = r.json()
    if "errors" in body:
        raise RuntimeError(body["errors"])
    return body["data"]


def main():
    token = begin_session()

    query = """
    query($wo:String!){
      workOrders(pageSize:1,
                 filter:{ workOrderNumber: [$wo] }){
        records{
          workOrderNumber
          part{ partNumber }
        }
      }
    }
    """

    data = gql(query, {"wo": TARGET_WO}, token)["workOrders"]["records"]

    if not data:
        print(f"No results for WO {TARGET_WO}")
        return

    wo_num  = data[0]["workOrderNumber"]
    part_no = data[0]["part"]["partNumber"] if data[0]["part"] else "—"

    print(f"{wo_num}  →  {part_no}")


if __name__ == "__main__":
    try:
        main()
    except Exception as e:
        print("Error:", e)
        sys.exit(1)
